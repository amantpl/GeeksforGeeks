#include <bits/stdc++.h>

using namespace std;

class Solution{
public:
	
	void printTriangle(int n) {
	    int counter=n;
	    for(int i=0;i<n;i++){
	        
	        for(int j=counter;j>0;j--){
	            cout<<counter-j+1<<" ";
	        }
	        counter--;
	        cout<<endl;
	    }
	}
};

//{ Driver Code Starts.

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        
        Solution ob;
        ob.printTriangle(n);
    }
    return 0;
}
