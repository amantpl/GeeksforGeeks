class Solution {
  public:
    void printSquare(int n) {
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i!=n-1 && i!=0 && j!=0 && j!=n-1){
                cout<<" ";
            }else{
                cout<<"*";
            }
            
            }
            
            cout<<endl;
        }
    }
};

//{ Driver Code Starts.

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;

        Solution ob;
        ob.printSquare(n);
    }
    return 0;
}
